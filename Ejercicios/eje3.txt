Dado un arreglo, encuentra el número más frecuente usando un diccionario.
public static int moda(int[] v) {
 Map<Integer, Integer> frecuencia = new HashMap<>();
 int maxFrecuencia = 0;
 int moda = v[0];
 for (int num : v) {
 int f = frecuencia.getOrDefault(num, 0) + 1;
 frecuencia.put(num, f);
 if (f > maxFrecuencia) {
 maxFrecuencia = f;
 moda = num;
 }
 }
 return moda;
}
